Google Sheets setup

Connect both partners to a shared database in 5 minutes

This uses Google Apps Script as a free backend — no server needed. Both partners access the same Google Sheet in real time.
1

Create a new Google Sheet

Go to sheets.new and create a blank sheet. Name it "AG Lending Database". Share it with your partner's Google account with Editor access.

2

Create the 3 required tabs

At the bottom of the sheet, create 3 tabs named exactly: Borrowers, Loans, Payments

Add these headers in row 1 of each tab:

Borrowers: id | name | contact | address | date_added
Loans:     id | borrower_id | borrower_name | principal | interest_rate | penalty_rate | start_date | due_date | status | notes
Payments:  id | loan_id | borrower_name | date | amount | penalty_applied | interest_applied | principal_applied | balance_after | notes
3

Open Apps Script editor

In your Google Sheet, go to Extensions → Apps Script. Delete any existing code and paste the script below:

const SHEET_ID = 'YOUR_SHEET_ID_HERE'; // Replace with your sheet ID from the URL

function doGet(e) {
  const sheet = SpreadsheetApp.openById(SHEET_ID);
  const tab = e.parameter.tab;
  const data = sheet.getSheetByName(tab).getDataRange().getValues();
  const headers = data[0];
  const rows = data.slice(1).map(r => {
    const obj = {};
    headers.forEach((h, i) => obj[h] = r[i]);
    return obj;
  });
  return ContentService.createTextOutput(JSON.stringify(rows))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  const payload = JSON.parse(e.postData.contents);
  const sheet = SpreadsheetApp.openById(SHEET_ID);
  const tab = sheet.getSheetByName(payload.tab);
  if (payload.action === 'append') {
    tab.appendRow(payload.row);
  } else if (payload.action === 'update') {
    const data = tab.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][0]) === String(payload.id)) {
        payload.row.forEach((val, col) => { if (val !== null) tab.getRange(i+1, col+1).setValue(val); });
        break;
      }
    }
  } else if (payload.action === 'delete') {
    const data = tab.getDataRange().getValues();
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][0]) === String(payload.id)) { tab.deleteRow(i+1); break; }
    }
  }
  return ContentService.createTextOutput(JSON.stringify({status:'ok'}))
    .setMimeType(ContentService.MimeType.JSON);
}

Replace YOUR_SHEET_ID_HERE with your sheet's ID — it's the long string in the URL between /d/ and /edit.

4

Deploy as web app

Click Deploy → New deployment → Web app. Set:
— Execute as: Me
— Who has access: Anyone (so the app can call it without login)
Then click Deploy and copy the web app URL.

5

Paste the URL in Settings

Go to the Settings tab in this app and paste the Apps Script web app URL. Click Save. Both you and your partner paste the same URL — you'll both be writing to the same sheet.

⚠ The Apps Script URL acts as your API key — anyone with it can read/write your data. Keep it private. Don't share it publicly or commit it to a public GitHub repo.
